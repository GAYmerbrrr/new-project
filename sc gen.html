function search() { var keyword = document.getElementById("searchBox").value.toLowerCase(); // Ambil nilai dari kotak pencarian dan konversi menjadi huruf kecil var content = document.getElementById("content").innerHTML.toLowerCase(); // Ambil konten yang ingin dicari if (content.includes(keyword)) { // Periksa apakah konten mengandung kata kunci // Redirect ke halaman baru dengan hasil pencarian window.location.href = "hasil_pencarian.html?keyword=" + keyword; } else { alert("Konten tidak ditemukan."); // Tampilkan pesan jika konten tidak ditemukan } } Hasil Pencarian
